194 NV_INLINE void Nv::Blast::TkGroup::process()
195 {
196  uint32_t jobCount = startProcess();
197  if (jobCount > 0)
198  {
199  TkGroupWorker* worker = acquireWorker();
200  for (uint32_t i = 0; i < jobCount; i++)
201  {
202  worker->process(i);
203  }
204  returnWorker(worker);
205  }
206  endProcess();
207 }
