117  uint32_t getOpposite(uint32_t a, uint32_t b)
118  {
119  if (ea != a && ea != b)
120  return ea;
121  if (eb != a && eb != b)
122  return eb;
123  if (ec != a && ec != b)
124  return ec;
125  return NOT_VALID_VERTEX;
126  }
127 
128  bool isContainEdge(uint32_t a, uint32_t b)
129  {
130  return (a == ea || a == eb || a == ec) && (b == ea || b == eb || b == ec);
131  }
