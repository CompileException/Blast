00650         q[0] = a * k;
00651         q[1] = -n[0] * p[2];
00652         q[2] = n[0] * p[1];
00653     }
00654     else {
00655         // choose p in x-y plane
00656         btScalar a = n[0] * n[0] + n[1] * n[1];
00657         btScalar k = btRecipSqrt(a);
00658         p[0] = -n[1] * k;
00659         p[1] = n[0] * k;
00660         p[2] = 0;
00661         // set q = n x p
00662         q[0] = -n[2] * p[1];
00663         q[1] = n[2] * p[0];
00664         q[2] = a * k;
00665     }
00666 }
