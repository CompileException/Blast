277 {

283 NV_INLINE uint32_t Asset::getHierarchyCount() const
284 {
285  const NvBlastChunk* chunks = getChunks();
286  for (uint32_t i = 0; i < m_chunkCount; ++i)
287  {
288  if (!isInvalidIndex(chunks[i].parentChunkIndex))
289  {
290  return i;
291  }
292  }
293  return m_chunkCount;
